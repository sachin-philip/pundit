from pundit.base import PunditBase
from pundit.condition import ConditionGroup
from pundit import Pundit

base = PunditBase()
base.add_structure("designation", "rank")
base.structure


conditions = ConditionGroup()
conditions.add_condition("chief","IN", "Success", "Failed")
conditions.add_condition("engineer","IN", "Success", "Failed")
conditions.conditions


pundit = Pundit(base, conditions)

pundit.evaluate('chief technology officer')

>>> Success

pundit.evaluate('sofware engineer')

>>> Success

pundit.evaluate('sofware')

>>> Failed
